Python

for Simple stock Analysis

By Stchanga 30 May 2021

簡易即時link url 取得台灣證卷交易所資訊

並用python 做一個股票即時的分析

練習demo panda dataframe

import requests
import pandas as pd

url='https://www.twse.com.tw/exchangeReport/BWIBBU_d?response=csv&date=20210907&selectType=ALL'
data=requests.get(url)
#print(data.text.replace("\"","")) #字串多雙引號，可replace""
data_list=[]
for row in data.text.replace("\"","").split("\n"):
    #print(len(row,split(","))) #可發現切出8個的 為股票資料
    rows=row.split(",");
    if len(rows) == 8 and rows[0] != '證券代號':
        
        d = {
            '證券代號':rows[0],
            '證券名稱':rows[1],
            '殖利率':None if rows[2]=='-' else float(rows[2]),
            '股利年度':rows[3],
            '本益比':None if rows[4]=='-' else float(rows[4]),
            '股價淨值比':None if rows[5]=='-' else float(rows[5]),
            '財報年 / 季':rows[6],
        
        }
        data_list.append(d)

#print(data_list)

df = pd.DataFrame(data_list)
#print(df.head())
i=df.get('殖利率')>10
j=df.get('本益比')<10
k=df.get('股價淨值比').between(0,1)
print(df[i & j & k])
