DEMO Google 報表 做雲端自動化

當Google表單，填表完成，自動通知

說明:即時雲端報表，能即時通知損壞設備

論述:需設置line notify 並使用javascript動態判斷表格透過line notify即時通知LLINE群組

var lineToken='line Notify權杖碼';
var LineText='';

function main(){
  var sheet = SpreadsheetApp.getActiveSheet();
  var range=sheet.getDataRange();
  var data = range.getValues();

  for(var i=1;i<data.length;++i){
    var sendText=sheet.getRange(i+1,8).getValues();
    if(sendText==''){
      var rows=data[i];
      var Time=Utilities.formatDate(rows[0],'GMT+8','yyyy/MM/dd HH:mm');
      var Depart=rows[1];
      var Name=rows[2];
      var Location=rows[3];
      var Device=rows[4];
      var Desc=rows[5];
      var IMG=rows[6];
      sheet.getRange(i+1,8).setValue('已發佈');
      LineText='\n報修時間:'+Time+'\n\n'+
      '報修單位:'+Depart+'\n\n'+
      '報修者:'+Name+'\n\n'+
      '損壞地點:'+Location+'\n\n'+
      '損壞設備:'+Device+'\n\n'+
      '損壞狀況說明:'+Desc+'\n\n'+
      '損壞照片:'+IMG;
      //Logger.log(LineText);
      sendToLine(LineText);
    }
  }
}

function sendToLine(LineText){
  var token = lineToken;
  var URL ="https://drive.google.com/drive/folders/1g3rsxIMCb9uKLDQyS_Emu839spC6-3zaQ3AOpN8m6nQeokeNjf9ZKYUQS2c0soL-nWRXyJoC?usp=sharing"
  var options =
   {
     "method": "post",
     "headers" : {"Authorization" : "Bearer "+ token},
     "payload":{
        'message':'\n'+ LineText,
        'imageThumbnail': URL,
        'imageFullsize': URL
     }
   };
  UrlFetchApp.fetch("https://notify-api.line.me/api/notify", options);
}
